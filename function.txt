function myFunction() {
    console.log("Hello, I am a function!");
}

// Adding properties to the function (just like an object)
myFunction.name = "Utility Function";
myFunction.version = "1.0";
myFunction.isUtility = true;
myFunction.details = {
    category: "Math",
    usage: "Perform calculations"
};

// Adding a method to the function
myFunction.describe = function () {
    console.log(`Function Name: ${this.name}, Version: ${this.version}`);
};

// Accessing properties
console.log(myFunction.name);        // Output: Utility Function
console.log(myFunction.details.usage); // Output: Perform calculations

// Calling a function method
myFunction.describe();  // Output: Function Name: Utility Function, Version: 1.0

// Modifying properties
myFunction.version = "2.0";

// Adding a new property
myFunction.author = "John Doe";

// Deleting a property
delete myFunction.isUtility;

console.log(myFunction);
