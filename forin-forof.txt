// Initialization: let i = 0
for(let i = 0; i < 5; i++) { 
    // Condition: i < 5
    // Iteration: i++
    
    // Expression:
    console.log("This is loop number " + (i+1));
}
So the flow goes like this: Initialization -> Condition Check -> Execute Expression -> Iteration -> Condition Check -> Execute Expression -> Iteration -> â€¦ until the condition is false.






for(let i = 0; i < 3; i++) {
    for(let j = 0; j < 3; j++) {
        console.log("i: " + i + ", j: " + j);
    }
}




Initialization of Outer Loop
Condition Check of Outer Loop
If Outer Loop Condition is True:
Initialization of Inner Loop
Condition Check of Inner Loop
If Inner Loop Condition is True:
Execute Inner Loop Expression
Inner Loop Iteration
Go back to Condition Check of Inner Loop
If Inner Loop Condition is False:
Outer Loop Iteration
Go back to Condition Check of Outer Loop
If Outer Loop Condition is False:
End of Execution



function checkName(name) {
    switch (name) {
        case 'Basha':
            return 'Bro you are Basha';
        case 'Shaik':
            return 'Bro you are Shaik';
        case 'Mansoor':
            return 'Bro you are true';
        default:
            return 'Name not recognized';
    }
}

console.log(checkName('Mansoor')); // Output: Bro you are true
